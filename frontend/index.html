async function login() { ADMIN_KEY = document.getElementById('admin-key').value; if(!ADMIN_KEY) return tg.showAlert('Введи ключ!'); tg.showAlert('⏳ Подключение к серверу... (первый запуск может занять 30 сек)'); try { // Используем AbortController для таймаута 60 секунд const controller = new AbortController(); const timeoutId = setTimeout(() => controller.abort(), 60000); const res = await fetch(`${API}/api/admin/stats`, { headers: { 'x-admin-key': ADMIN_KEY }, signal: controller.signal }); clearTimeout(timeoutId); if(!res.ok) throw new Error('Неверный ключ'); document.getElementById('login-section').style.display = 'none'; document.getElementById('admin-panel').style.display = 'block'; refreshData(); } catch(e) { if(e.name === 'AbortError') { tg.showAlert('⏰ Таймаут! Сервер ещё спит. Подожди 30 сек и попробуй снова.'); } else { tg.showAlert('❌ ' + e.message); } } }
